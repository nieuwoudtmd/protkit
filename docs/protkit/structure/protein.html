217    def filter_chains(self, chain_criteria: Optional[List] = None) -> Iterator[Chain]:
218        """
219        Filters chains in the protein based on the specified criteria.
220        Yielded chains must match all criteria.
221
222        Args:
223            chain_criteria (list of tuple): The criteria to filter the chains.
224
225        Yields:
226            Chain: The next chain in the protein that matches the criteria.
227        """
228        if chain_criteria is None:
229            chain_criteria = []
230        for chain in self._chains.values():
231            match = True
232            for key, value in chain_criteria:
233                if type(value) is list:
234                    if chain.get_attribute(key) not in value:
235                        match = False
236                        break
237                elif chain.get_attribute(key) != value:
238                    match = False
239                    break
240            if match:
241                yield chain

243    def rename_chain(self, chain_id_from: str, chain_id_to: str) -> None:
244        """
245        Renames a chain in the protein.
246
247        Args:
248            chain_id_from (str): The ID of the chain to rename.
249            chain_id_to (str): The new ID of the chain.
250
251        Returns:
252            None
253        """
254        if chain_id_from not in self._chains:
255            raise Exception(f"Chain {chain_id_from} does not exist")
256        elif chain_id_to in self._chains:
257            raise Exception(f"Chain {chain_id_to} already exists")
258        else:
259            self._chains[chain_id_to] = self._chains[chain_id_from]
260            self._chains[chain_id_to].chain_id = chain_id_to
261            self._chains.pop(chain_id_from)
262
263    def remove_chains(self, chain_ids: Union[List[str], str]) -> None:
264        """
265        Removes chains from the protein.
266
267        Args:
268            chain_ids (list of str): The IDs of the chains to remove.
269
270        Returns:
271            None
272        """
273        if type(chain_ids) is str:
274            chain_ids = [chain_ids]
275        for chain_id in chain_ids:
276            if chain_id not in self._chains:
277                raise Exception(f"Chain {chain_id} does not exist")
278            else:
279                self._chains.pop(chain_id)

292            chain_ids = [chain_ids]
293        remove_chain_ids = list(set(self._chains.keys()) - set(chain_ids))
294        self.remove_chains(remove_chain_ids)
