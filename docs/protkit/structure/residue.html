129        Returns a deep copy of the residue, excluding the chain.

393    def add_atom(self, atom_id: str, atom: Atom) -> Atom:
394        """
395        Adds an atom to the residue.
396
397        Args:
398            atom_id (str): The atom's ID.
399            atom (Atom): The atom.
400
401        Returns:
402            Atom: The atom.
403        """
404        if atom_id in self._atoms:
405            current_atom = self._atoms[atom_id]
406            if current_atom.is_disordered and atom.is_disordered:
407                current_atom.merge_disordered_atom(atom)
408                return current_atom
409            else:
410                raise Exception("Atom ID {atom_id} already exists in residue.")
411        else:
412            self._atoms[atom_id] = atom
413            atom.residue = self
414            return atom

508    def filter_atoms(self, atom_criteria: Optional[List] = None) -> Iterator[Atom]:
509        """
510        Returns an iterator of atoms that match the specified criteria.
511
512        Args:
513            atom_criteria (Optional[List]): The criteria for selecting atoms.
514
515        Yields:
516            Atom: The next atom that matches the criteria.
517        """
518        if atom_criteria is None:
519            atom_criteria = []
520        for atom in self._atoms.values():
521            match = True
522            for key, value in atom_criteria:
523                if type(value) is list:
524                    if atom.get_attribute(key) not in value:
525                        match = False
526                        break
527                elif atom.get_attribute(key) != value:
528                    match = False
529                    break
530            if match:
531                yield atom
