311    def filter_residues(self, residue_criteria: Optional[List] = None) -> List[Residue]:
312        """
313        Yields residues that match the specified criteria.
314
315        Args:
316            residue_criteria (Optional[List]): List of residue criteria.
317
318        Yields:
319            Residue: The next residue that matches the specified criteria.
320        """
321        if residue_criteria is None:
322            residue_criteria = []
323        for residue in self._residues:
324            match = True
325            for key, value in residue_criteria:
326                if type(value) is list:
327                    if residue.get_attribute(key) not in value:
328                        match = False
329                        break
330                elif residue.get_attribute(key) != value:
331                    match = False
332                    break
333            if match:
334                yield residue

336    def seqres_analysis(self):
337        """
338        Compares the sequence of the chain to the SEQRES record.
339        """
340        sequence_residues = self.get_attribute("seqres")
341        structure_residues = [residue.residue_type for residue in self._residues if not residue.is_hetero]
342        alignment1, alignment2 = Sequence.align(sequence_residues, structure_residues)
343        return alignment1, alignment2

345    def renumber_residues(self, new_numbers: Optional[List[int]] = None) -> None:
346        """
347        Renumbers the residues in the chain. Residues are numbered
348        sequentially, starting at 1.  If new_numbers is not specified,
349        the residues are numbered sequentially, starting at 1.
350        Insertion codes are removed.
351
352        Args:
353            new_numbers (Optional[List[int]]): List of new residue numbers.
354                If not specified, the residues are numbered sequentially,
355                starting at 1.
356
357        Returns:
358            None
359
360        Raises:
361            None
362        """
363        if new_numbers is None:
364            new_numbers = list(range(1, len(self._residues) + 1))
365
366        for i in range(len(self._residues)):
367            self._residues[i].sequence_no = new_numbers[i]
368            self._residues[i].insertion_code = ""

370    def remove_hetero_residues(self, hetero_residue_types: Optional[Union[str, List[str]]] = None) -> None:
371        """
372        Removes hetero residues from the chain.
373
374        Args:
375            hetero_residue_types (Optional[Union[str, List[str]]]): The residue types to be removed.
376                If not specified, all hetero residues are removed.
377
378        Returns:
379            None
380        """
381        if hetero_residue_types is None:
382            self._residues = [residue for residue in self._residues if not residue.is_hetero]
383        else:
384            if type(hetero_residue_types) is str:
385                hetero_residue_types = [hetero_residue_types]
386            self._residues = [residue for residue in self._residues if residue.residue_type not in hetero_residue_types]

690    def sum_attribute(self, key: str) -> None:

697                residue.sum_attribute(key)
