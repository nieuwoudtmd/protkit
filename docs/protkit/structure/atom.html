 47        """
 48        Constructor for the Atom class.
 49
 50        Args:
 51            element (str): The element associated with the atom.
 52            atom_type (str): The atom type.
 53            x (float): The x coordinate of the atom.
 54            y (float): The y coordinate of the atom.
 55            z (float): The z coordinate of the atom.
 56            is_hetero (bool, optional): The hetero flag. Defaults to False.
 57            is_disordered (bool, optional): The disordered flag. Defaults to False.
 58            alt_loc (Optional[str], optional): The alt_loc identifier. Defaults to None.
 59            occupancy (Optional[float], optional): The occupancy. Defaults to None.
 60            temp_factor (Optional[float], optional): The temperature factor. Defaults to None.
 61            assigned_charge (Optional[str], optional): The assigned charge. Defaults to None.
 62            calculated_charge (Optional[float], optional): The calculated charge (PQR). Defaults to None.
 63            radius (Optional[float], optional): The radius (PQR). Defaults to None.
 64            residue (Optional[Residue], optional): The residue the atom forms part of. Defaults to None.
 65
 66        Returns:
 67            None
 68        """
 69
 70        # Core properties
 71        self._x: float = x
 72        self._y: float = y
 73        self._z: float = z
 74        self._element: str = element
 75        self._atom_type: str = atom_type
 76
 77        # It could happen that the element is not set. For example, PQR files
 78        # do not have an element field.
 79        # In this case, the element is set to the first character of the atom type.
 80        # This should generally work, although some programs use naming conventions for
 81        # atom types that are not standard, for example starting a hydrogen with
 82        # a numerical value rather than H. There may also be complications for
 83        # two-letter element names. For example "CA" could be Calcium or Alpha Carbon.
 84        # Further improvements should be made to handle these cases.
 85        if self._element is None or self._element == "":
 86            if self._atom_type is not None and self._atom_type != "":
 87                self._element = self._atom_type[0]
 88
 89        # Derived properties
 90        self._is_hetero: bool = is_hetero
 91        self._is_disordered: bool = is_disordered
 92
 93        # Optional attributes
 94        if alt_loc is not None and alt_loc != "":
 95            self.set_attribute("alt_loc", alt_loc)
 96            # If alt_loc is set, occupancy is also set
 97            # There are examples where alt_loc is set, even though occupancy is 1.0.
 98            # In these cases, the if statement below will not set the occupancy
 99            # to 1.0, which is why it needs to be set here.
100            self.set_attribute("occupancy", occupancy)
101        if occupancy is not None and occupancy != 1.0:
102            self.set_attribute("occupancy", occupancy)
103        if temp_factor is not None:
104            self.set_attribute("temp_factor", temp_factor)
105        if assigned_charge is not None and assigned_charge != "":
106            self.set_attribute("assigned_charge", assigned_charge)
107        if calculated_charge is not None:
108            self.set_attribute("calculated_charge", calculated_charge)
109        if radius is not None:
110            self.set_attribute("radius", radius)
111
112        # If alt_loc is specified, the atom is disordered
113        # Keep track of all disordered states.
114        if self.get_attribute("alt_loc") is not None:
115            self._is_disordered = True
116            self._disordered_states = [{
117                "alt_loc": self.get_attribute("alt_loc"),
118                "occupancy": self.get_attribute("occupancy"),
119                "temp_factor": self.get_attribute("temp_factor"),
120                "x": self._x,
121                "y": self._y,
122                "z": self._z
123            }]
124
125        self._residue: Optional[Residue] = residue

128        """
129        Returns a deep copy of the atom, excluding the residue.

133        """
134        residue = self._residue
135        self._residue = None
136        atom = deepcopy(self)
137        self._residue = residue
138        return atom

390    def merge_disordered_atom(self, atom: Atom) -> None:
391        """
392        Merges the disordered atom with the current atom.
393        Keep track of the additional disordered information
394        from the atom that the merge is done with.
395
396        Args:
397            atom (Atom): The atom to merge with.
398
399        Returns:
400            None
401        """
402        self._disordered_states.append({
403            "alt_loc": atom.get_attribute("alt_loc"),
404            "occupancy": atom.get_attribute("occupancy"),
405            "temp_factor": atom.get_attribute("temp_factor"),
406            "x": atom.x,
407            "y": atom.y,
408            "z": atom.z
409        })
410        self._is_disordered = True
411
412    def fix_disordered_atom(self) -> None:
413        """
414        Fixes the disordered atom by selecting the disordered
415        state with the highest occupancy.
416
417        Returns:
418            None
419        """
420        if self._is_disordered:
421            # Sort disordered states by occupancy
422            self._disordered_states.sort(key=lambda dis_state: dis_state["occupancy"], reverse=True)
423
424            # Select the disordered state with the highest occupancy
425            state = self._disordered_states[0]
426
427            # Set the atom's properties to the disordered state with the highest occupancy
428            self.set_attribute("alt_loc", state["alt_loc"])
429            self.set_attribute("occupancy", state["occupancy"])
430            self.set_attribute("temp_factor", state["temp_factor"])
431            self._x = state["x"]
432            self._y = state["y"]
433            self._z = state["z"]
434
435            # Remove the disordered states
436            self._disordered_states = None
437            self._is_disordered = False
